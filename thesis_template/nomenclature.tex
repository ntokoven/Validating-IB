
\usepackage{nomencl}
\makenomenclature
\setlength{\nomitemsep}{-\parsep}

%% Creates the groups for the nomenclature
% -----------------------------------------
\usepackage{ifthen}
\renewcommand{\nomgroup}[1]{%
	\item[\bfseries
		\ifthenelse{\equal{#1}{G}}{Groups}{%
		\ifthenelse{\equal{#1}{O}}{Operators}{%
		\ifthenelse{\equal{#1}{S}}{Other Symbols}{%
		\ifthenelse{\equal{#1}{N}}{Notation}{%
		}}}}%
	]}

% -----------------------------------------

% This will add the units
%----------------------------------------------
\newcommand{\nomunit}[1]{%
	\renewcommand{\nomentryend}{\hspace*{\fill}#1}}
%----------------------------------------------

\nomenclature[N]{$\Pr$}{propability function}
\nomenclature[N]{$\EX$}{expectation}
\nomenclature[N]{$\Cov$}{covariance}
\nomenclature[N]{$\vvec$}{a vector}
